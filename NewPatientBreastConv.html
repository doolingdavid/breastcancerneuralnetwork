def get_rates():
    """Need to loop through threshold values high to low, accumulate
    the true positive rates and false positive rates into a list
    and then turn it all into three arrays and return them."""
    
    fprlist = []
    tprlist = []
    
    ctClass = [-0.01 + i*0.01 for i in range(103)][::-1]
    
    duh = list(np.arange(1.1, -.01, -.01))
    
    duh = ctClass
    
    for i in range(len(duh)):
        print(duh[i], type(float(duh[i])))
        prob0pos = (dfroc['pp0'] >= float(duh[i]))  # this value here is the threshold
        prob0neg = (dfroc['pp0'] < float(duh[i]))
        surv0pos = (dfroc['true0'] == 1)
        surv0neg = (dfroc['true0'] != 1)
        
        tp = dfroc[prob0pos & surv0pos].shape[0]
        fp = dfroc[prob0pos & surv0neg].shape[0]
        tn = dfroc[prob0neg & surv0neg].shape[0]
        fn = dfroc[prob0neg & surv0pos].shape[0]
        
        accuracy = (tp + tn) / float(tp + fp + tn + fn)
        
        
        #print "predictions: "
        
        #print "accuracy: ", accuracy
        
        tpr = tp/float(tp + fn)
        
        #print "True Positive rate: ", tpr
        
        tprlist.append(tpr)
        
        fpr = fp/float(fp + tn)
        
        #print "False positive rate: ", fpr
        
        fprlist.append(fpr)
        
        print("Confusion matrix:")
        print(tp, fn)
        print(fp, tn)
        
    
        
    return tprlist, fprlist

def get_survival_function(patient, df, clf):
    """Takes as input the value of newindex and returns
    an array corresponding to the cdf."""
    import operator
    import math
    dfcopy = df.loc[patient].copy()
    
    if type(dfcopy) != pd.core.series.Series:
        print(df[['month','newtarget']].loc[patient].tail(1))
        print(df[['month']].loc[patient].tail(1).values[0][0])
    else:
        print("Month: ", df[['month']].loc[patient].values[0])
    newtargetvalue = df[['newtarget']].loc[patient].tail(1).values[0]
    if newtargetvalue == 1:
        if type(dfcopy) != pd.core.series.Series:
            dfcopy_data = dfcopy[list(df.columns)[:-2]].iloc[0].values
        else:
            dfcopy_data = dfcopy.values[:-2]
        A = []
        p_so_far = 1
        for i in range(100):
            thing = np.append(dfcopy_data,i)
            p_cur = clf.predict_proba(thing[None,...],verbose=0)[0][1]
            A.append(p_so_far * p_cur)
            p_so_far = p_so_far*(1 - p_cur)
    
    
        index, value = max(enumerate(A), key=operator.itemgetter(1))
        As = pd.Series(A)
        
        if type(dfcopy) != pd.core.series.Series:
            absoluteerror = abs(index - df[['month']].loc[patient].tail(1).values[0][0])
        else:
            absoluteerror = abs(index - df[['month']].loc[patient]).values[0]
            
            
            
         
        print("Prediction: ", index, " with probability: ", value)
        print("absolute error: ", absoluteerror)
    
        As.plot()
        plt.title("PMF for Patient " + str(patient))
        
        Asurv = 1 - As.cumsum()
        
        Asurv.plot()
        plt.title("Survival Function for Patient " + str(patient))
    
        return Asurv
    
    else:
        print("not using that one", patient)
